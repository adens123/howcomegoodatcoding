String Attributes and Methods

// length
let str = 'Jacky Wu';
console.log('---------');
console.log('變數str的長度是：', str.length);

// [n] index
let str1 = 'My name is Jacky Wu';
console.log('----------');
console.log('str1的第4個字是：', str1[3]);
console.log('str1的最後一個字是：', str1[str1.length - 1]);
console.log('str1的第100個字是：', str1[99]);

// slice(indexStart [, indexEnd])
let str2 = 'Iamlearningjavascript';
console.log('----------');
console.log('str2從第4個字開始切：', str2.slice(3));
console.log('str2從第1個字開始切到第10個字：', str2.slice(0, 11));

// indexOf(substring) if true return index, else return -1
let str3 = 'wearefamily';
console.log('----------');
console.log(
  'str3是否包含are，有return index，沒有return -1：',
  str3.indexOf('are')
);
console.log(
  'str3是否包含ccc，有return index，沒有return -1：',
  str3.indexOf('ccc')
);

// toUpperCase(), toLowerCase()
let str4 = 'Uppercaseandlowercase';
console.log('----------');
console.log('str4轉換大寫，原str4不變：', str4.toUpperCase(), str4);
console.log('str4轉換小寫，原str4不變：', str4.toLowerCase(), str4);
str4 = str4.toUpperCase();
console.log('str4轉換大寫，並取代原str4', str4);

// split() return an array
let sentance = 'Today is a good day';
let result = sentance.split(' ');
console.log('sentance分拆為單字：', sentance.split(' '));

// startsWith(), endsWith(), includes(), charCodeAt(index)
let sentance1 = 'I am a good student';
console.log('sentance1開頭是否包含good：', sentance1.startsWith('I'));
console.log('sentance1開頭是否包含Good：', sentance1.startsWith('i'));
console.log('sentance1結尾是否包含student：', sentance1.endsWith('student'));
console.log('sentance1結尾是否包含Student：', sentance1.endsWith('Student'));
console.log('sentance1是否包含am：', sentance1.includes('am'));
console.log('sentance1是否包含are：', sentance1.includes('are'));
console.log('sentance1的index 0的uft-16的編碼：', sentance1.charCodeAt(0));
